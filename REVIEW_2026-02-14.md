import anthropic

class LLMRunbookEvolution:
    def __init__(self):

        prompt = f"""
Analyze this incident and suggest runbook improvements:

**Time to Resolve:** {incident.duration_minutes} minutes

{incident.runbook.steps}

**Similar Past Incidents:**
{self.get_similar_incidents(incident)}

Suggest 3-5 specific improvements to the runbook.
Format each as: [TYPE] [ACTION] [REASONING]
Types: ADD_STEP, REMOVE_STEP, MODIFY_STEP, NEW_SECTION
"""
        
        response = self.client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=1500,
            messages=[{"role": "user", "content": prompt}]
        )
        
        return self.parse_suggestions(response.content[0].text)

#### 5. **Git-Backed Runbook Versioning**
Every change tracked, auditable, reversible:

```python
# version_control.py
from git import Repo

class RunbookVersionControl:
    def __init__(self, repo_path: str):
        self.repo = Repo(repo_path)
    
    def commit_annotation(self, runbook_path: str, annotation: dict):
        """Commit annotation with structured message"""
        # Append to runbook
        self.append_annotation(runbook_path, annotation)
        
        # Create commit
        self.repo.index.add([runbook_path])
        
        commit_msg = f"""incident: {annotation['incident_id']}

cause: {annotation['root_cause'][:50]}
fix: {annotation['fix_applied'][:50]}
time_to_resolve: {annotation['duration']}min
"""
        
        self.repo.index.commit(
            commit_msg,
            author=Actor("runbook-bot", "bot@runbooks.local")
        )
    
    def get_runbook_history(self, runbook_path: str) -> List[Commit]:
        """Get all changes to a runbook"""
        return list(self.repo.iter_commits(paths=runbook_path))
```
